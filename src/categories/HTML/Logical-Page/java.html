Java : Area of Circle

Input
import java.util.Scanner; public class AreaOfCircle { public static void main(String[] args) { Scanner scanner = new Scanner(System.in); int radius = 10; double pi = 3.14; double area = pi * radius * radius; System.out.printf("Area of circle is: %.2f", area); scanner.close(); } }
